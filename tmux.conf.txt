set -g window-status-current-style fg=blue,bg=black,bold
set -g status-style fg=white,bg=colour240

bind x set -g window-status-current-style fg=blue,bg=black,bold \; set -g prefix C-b
bind y set -g window-status-current-style fg=green,bg=black,bold \; set -g prefix C-y

bind c new-window -c "#{pane_current_path}"
bind | split-window -h
bind - split-window -v
bind r command-prompt "rename-window %%"
bind C-h swap-window -t -1
bind C-l swap-window -t +1
bind . source-file ~/.tmux.conf \; display-message ".tmux.conf reloaded"

bind h resize-pane -L 1
bind j resize-pane -D 1
bind k resize-pane -U 1
bind l resize-pane -R 1

set -g status-interval 1
set -g status-left " "
set -g status-right "%H:%M:%S #[bold]#h#[nobold] "
set -g window-status-format "#I:#W "

set -g history-limit 65536
set -g base-index 1
set -g pane-base-index 1
set -g renumber-windows on

set -g default-shell $SHELL
set -g default-terminal screen-256color
set -g terminal-overrides 'xterm:colors=256'
set-environment -g CHERE_INVOKING 1
